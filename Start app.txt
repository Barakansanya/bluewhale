# Restart PostgreSQL
sudo service postgresql restart

# Backend setup
cd server
npm install
npx prisma migrate dev --name init
npx prisma generate
npm run prisma:seed
npm run dev

#Real data sync
# Sync a single company
curl -X POST http://localhost:5000/api/v1/sync/company/APN

# Sync all companies (takes ~15 seconds)
curl -X POST http://localhost:5000/api/v1/sync/all

# Frontend setup (new terminal)
cd ../client
npm install
npm run dev
